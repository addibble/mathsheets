\subsection*{Exponential Rules}
\begin{flushleft}
$ a^m \cdot a^n = a^{m+n} $ \\
$ \dfrac{a^m}{a^n} = a^{m-n} $ \\
$ (a^m)^n = a^{mn} $ \\
$ a^0 = 1 \ (a \ne 0) $ \\
$ a^{-n} = \dfrac{1}{a^n} $ \\
$ a^{1/n} = \sqrt[n]{a} $
\end{flushleft}

\subsection*{Logarithmic Rules}
\begin{flushleft}
$ \log_b(xy) = \log_b x + \log_b y $ \\
$ \log_b\left(\dfrac{x}{y}\right) = \log_b x - \log_b y $ \\
$ \log_b(x^r) = r \log_b x $ \\
$ \log_b b = 1 $ \\
$ \log_b 1 = 0 $ \\
$ \log_b x = \dfrac{\log_k x}{\log_k b} $ \\
$ b^{\log_b x} = x $ \\
$ \log_b(b^x) = x $
\end{flushleft}
\end{multicols}
